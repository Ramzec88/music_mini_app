function validateField(fieldId, isValid, message = '') { const field = document.getElementById(fieldId); const container = field.parentNode; // Remove existing validation message const existingMsg = container.querySelector('.validation-message'); if (existingMsg) existingMsg.remove(); // Update field styling if (isValid) { field.style.borderColor = '#48bb78'; } else { field.style.borderColor = '#f56565'; if (message) { const msg = document.createElement('div'); msg.className = 'validation-message'; msg.style.cssText = ` color: #f56565; font-size: 0.9rem; margin-top: 5px; `; msg.textContent = message; container.appendChild(msg); } } }
